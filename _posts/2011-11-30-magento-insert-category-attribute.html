Another Magento snippet. How to insert a new category attribute directly into the Magento database. 

insert into eav_attribute (
	attribute_id,
	entity_type_id,
	attribute_code,
	backend_type,
	frontend_input,
	frontend_label
) values (
	null,
	9,
	"extra_description",
	"text",
	"textarea",
	"Extra Description"
);

insert into eav_entity_attribute values (null, 9, 12, 7, 965, 28);
